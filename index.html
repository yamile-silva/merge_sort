								PSEUDOCODIGO MERGESORT 

							Funcion merge(A[],inicio, fin)
							|	Si(inicio < fin)
							|	|	med <- (inicio + fin)/2
							|	|	merge(inicio, med)
							|	|	merge(med+1, fin)
							|	|	insertar <- inicio
							|	|	izq <- inicio
							|	|	der <- med+1 
							|	|	Mientras (insertar <= fin)
							|	|	|	Si (izq  fin o A[izq ] <= A[der])
							|	|	|	|	AUX[insertar] <- A[izq ]
							|	|	|	|	insert <- insert + 1 
							|	|	|	|	izq <- izq +1
							|	|	|	Sino 
							|	|	|	|	AUX[insertar] <- A[der]
							|	|	|	|	insert <- insert + 1
							|	|	|	|	der <- der +1 
							|	|	|	Fin_Si
							|	|	Fin_Mientras
							|	|	Para insertar desde inicio hasta fin
							|	|	|	A[insertar] <- AUX[insert]
							|	|	Fin_Para
							|	Sino 											ESTUDIANTES:
							|	|	break										JOSE LUIS VARILLAS MELENDEZ
